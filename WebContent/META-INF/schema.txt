use swimbusan;

create table member(
	userId varchar(20) primary key,
    userPw varchar(20) not null,
    userName varchar(20) not null,
    userEmail varchar(50) not null,
    userZipcode int,
    userAddress1 varchar(50),
    userAddress2 varchar(50),
    userPhone varchar(20),
    userRegdate date
);

create table board(
	boardNum int primary key,
	userId varchar(20),
	userName varchar(20),
	boardPw varchar(20),
	boardSubject varchar(50) not null,
	boardContent varchar(2000) not null,
	boardFile varchar(50),
	boardRe_ref int,
	boardRe_lev int,
	boardRe_seq int,
	boardCount int,
	boardDate datetime,
	boardIp varchar(20),
    boardCategory varchar(20)
);

create table info(
	boardNum int primary key,
	userId varchar(20),
	userName varchar(20),
	boardPw varchar(20),
	boardSubject varchar(50) not null,
	boardContent varchar(2000) not null,
	boardFile varchar(50),
	boardRe_ref int,
	boardRe_lev int,
	boardRe_seq int,
	boardCount int,
	boardDate datetime,
	boardIp varchar(20),
    boardCategory varchar(20)
);

create table review(
	boardNum int primary key,
	userId varchar(20),
	userName varchar(20),
	boardPw varchar(20),
	boardSubject varchar(50) not null,
	boardContent varchar(2000) not null,
	boardFile varchar(50),
	boardRe_ref int,
	boardRe_lev int,
	boardRe_seq int,
	boardCount int,
	boardDate datetime,
	boardIp varchar(20),
    boardCategory varchar(20)
);

create table pool(
	poolNum int primary key,
	poolName varchar(20) not null unique,
    poolZipcode int,
    poolAddress1 varchar(50),
    poolAddress2 varchar(50),
    poolTel varchar(20),
    poolUrl varchar(50),
    poolContent varchar(2000),
    poolDate datetime
);

create table gallery(
	boardNum int primary key,
	userId varchar(20),
	userName varchar(20),
	boardPw varchar(20),
	boardSubject varchar(50) not null,
	boardContent varchar(2000) not null,
	boardFile varchar(200),
	boardRe_ref int,
	boardRe_lev int,
	boardRe_seq int,
	boardCount int,
	boardDate datetime,
	boardIp varchar(20),
    boardCategory varchar(20)
);
